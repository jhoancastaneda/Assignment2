CREATE TABLE CLASE_VEHICULO(
ID_CLASE_VEHICULO NUMBER  PRIMARY KEY, 
CLASE VARCHAR2(100),
CALIFICACION_VEHICULO INT
)

CREATE SEQUENCE CLASE_VEHICULO_SEQUENCE 
  START WITH 1
  INCREMENT BY 1;

CREATE TABLE VEHICULO (
REGISTRO_VEHICULO  NUMBER PRIMARY KEY, 
PLACA VARCHAR2(10),
ESTADO VARCHAR2(50),
KILOMETRAJE NUMBER(6,0),
MODELO NUMBER(4,0),
ID_CLASE_VEHICULO NUMBER REFERENCES CLASE_VEHICULO(ID_CLASE_VEHICULO),
VALOR_ACTUAL NUMBER(10,0),
VALOR_REEMPLAZO NUMBER(10,0),
FECHA_COMPRA DATE,
FECHA_PAGO_IMPUESTO DATE,
NUMERO_SEGURO VARCHAR2(50),
FECHA_PERDIDA DATE,
FECHA_EMISION DATE
)

CREATE SEQUENCE VEHICULO_SEQUENCE 
  START WITH 1
  INCREMENT BY 1;
  
CREATE TABLE CONDUCTOR(
NUMERO_EMPLEADO NUMBER PRIMARY KEY, 
NOMBRE_COMPLETO VARCHAR2(100),
CALIFICAION_CONDUCTOR INT
)

CREATE SEQUENCE CONDUCTOR_SEQUENCE 
  START WITH 1
  INCREMENT BY 1;

CREATE TABLE VEHICULO_CONDUCTOR(
ID_VEHICULO_CONDUCTOR NUMBER PRIMARY KEY, 
REGISTRO_VEHICULO NUMBER REFERENCES VEHICULO(REGISTRO_VEHICULO),
NUMERO_EMPLEADO NUMBER REFERENCES CONDUCTOR(NUMERO_EMPLEADO),
FECHA_ASIGNACION DATE,
ESTADO_CALIFICACION VARCHAR(50)
)

CREATE SEQUENCE VEHICULO_CONDUCTOR_SEQUENCE 
  START WITH 1
  INCREMENT BY 1;

CREATE TABLE SERVICIO(
ID_SERVICIO NUMBER PRIMARY KEY, 
NOMBRE VARCHAR2(100),
DESCRIPCION VARCHAR2(100)
)

CREATE SEQUENCE SERVICIO_SEQUENCE 
  START WITH 1
  INCREMENT BY 1;
  
CREATE TABLE VEHICULO_SERVICIO(
ID_VEHICULO_SERVICIO NUMBER PRIMARY KEY, 
REGISTRO_VEHICULO NUMBER REFERENCES VEHICULO(REGISTRO_VEHICULO),
ID_SERVICIO NUMBER REFERENCES SERVICIO(ID_SERVICIO),
ESTADO VARCHAR2(50),
FECHA DATE
)

CREATE SEQUENCE VEHICULO_SERVICIO_SEQUENCE 
  START WITH 1
  INCREMENT BY 1;
  
CREATE TABLE RECLAMACION_SEGURO(
ID_RECLAMACION_SEGURO NUMBER PRIMARY KEY, 
REGISTRO_VEHICULO NUMBER REFERENCES VEHICULO(REGISTRO_VEHICULO),
NATURALEZA_PAGO VARCHAR2(50),
COSTOS_REPARACION NUMBER(10,0)
)

CREATE SEQUENCE RECLAMACION_SEGURO_SEQUENCE 
  START WITH 1
  INCREMENT BY 1;

CREATE TABLE DETALLE_RECLAMACION(
ID_DETALLE_RECLAMACION NUMBER PRIMARY KEY, 
ID_RECLAMACION_SEGURO NUMBER REFERENCES RECLAMACION_SEGURO(ID_RECLAMACION_SEGURO),
FECHA DATE,
DESCRIPCION VARCHAR2(50),
CANTIDAD INT,
COSTO NUMBER(10,0),
ESTADO VARCHAR(50)
)

CREATE SEQUENCE DETALLE_RECLAMACION_SEQUENCE 
  START WITH 1
  INCREMENT BY 1;

--PUNTO 4 CREACION DE VISTA PARA PERFIL FINANCIERO

CREATE OR REPLACE VIEW VW_FINANCE_PROFILE AS
SELECT 
VEHICULO.REGISTRO_VEHICULO ,
VEHICULO.KILOMETRAJE,
VEHICULO.MODELO,
VEHICULO.VALOR_ACTUAL,
CLASE_VEHICULO.CLASE AS CLASE_VEHICULO,
VEHICULO.VALOR_REEMPLAZO,
VEHICULO.ESTADO,
CLASE_VEHICULO.CALIFICACION_VEHICULO,
(SELECT COUNT(*) FROM VEHICULO_SERVICIO WHERE VEHICULO_SERVICIO.REGISTRO_VEHICULO=VEHICULO.REGISTRO_VEHICULO) AS NUMERO_DE_SERVICIOS,
CASE WHEN (SELECT SUM(COSTOS_REPARACION) FROM RECLAMACION_SEGURO WHERE RECLAMACION_SEGURO.REGISTRO_VEHICULO=VEHICULO.REGISTRO_VEHICULO)  IS NULL THEN 0 ELSE
(SELECT SUM(COSTOS_REPARACION) FROM RECLAMACION_SEGURO WHERE RECLAMACION_SEGURO.REGISTRO_VEHICULO=VEHICULO.REGISTRO_VEHICULO)
END AS TOTAL_RECLAMACION_SEGURO
FROM VEHICULO
INNER JOIN CLASE_VEHICULO ON CLASE_VEHICULO.ID_CLASE_VEHICULO=VEHICULO.ID_CLASE_VEHICULO;


SELECT * FROM VW_FINANCE_PROFILE;


--PUNTO 5 CREACION DE FUNCION PARA NOMBRE SERVICIO DE ACUERDO AL KILOMETRAJE

CREATE OR REPLACE FUNCTION FN_NOMBRE_SERVICIO(
 REGISTRO_VEHICULO2 NUMBER)
RETURN VARCHAR2 IS
 NOMBRE_SERVICIO VARCHAR2(50);
 REGISTRO INT;
 KIL NUMBER;
 VAL INT;
 VAL2 INT;
BEGIN
 NOMBRE_SERVICIO:='';
 SELECT COUNT(*) INTO REGISTRO FROM VEHICULO_SERVICIO WHERE REGISTRO_VEHICULO=REGISTRO_VEHICULO2;
 IF(REGISTRO<>0) THEN 
    SELECT KILOMETRAJE INTO KIL FROM VEHICULO WHERE REGISTRO_VEHICULO=REGISTRO_VEHICULO2;
    VAL:=FLOOR(KIL/3000);
    VAL2:=0;
    FOR i IN 1..VAL LOOP
            VAL2:=VAL2+3000;
         END LOOP;
    NOMBRE_SERVICIO:=VAL2||'_service';
 END IF;
 RETURN NOMBRE_SERVICIO;
END;

--PUNTO 6 CREACION DE PROCEDIMIENTO ALMACENADO PARA INSERTAR SERVICIO Y VEHICULO_SERVICIO

CREATE OR REPLACE PROCEDURE PR_INSERTAR_SERVICIO (
 REGISTRO_VEHICULO2 NUMBER) IS
 REGISTRO VARCHAR2(50);
 DES SERVICIO.DESCRIPCION%TYPE;
 INSERTA NUMBER;
 KIL NUMBER;
 BEGIN
    SELECT FN_NOMBRE_SERVICIO(REGISTRO_VEHICULO2) INTO REGISTRO FROM DUAL; 
    IF(REGISTRO IS NOT NULL OR REGISTRO<>'') THEN
        SELECT KILOMETRAJE INTO KIL FROM VEHICULO WHERE REGISTRO_VEHICULO=REGISTRO_VEHICULO2;
        DES:='Mandatory '||REGISTRO||' service, current mileage '||KIL;
        INSERT INTO SERVICIO (ID_SERVICIO,NOMBRE,DESCRIPCION) VALUES (SERVICIO_SEQUENCE.NEXTVAL,REGISTRO,DES);
        COMMIT;
        
        SELECT ID_SERVICIO INTO INSERTA FROM SERVICIO WHERE NOMBRE=REGISTRO;

        INSERT INTO VEHICULO_SERVICIO (ID_VEHICULO_SERVICIO,REGISTRO_VEHICULO,ID_SERVICIO,ESTADO,FECHA) VALUES (VEHICULO_SERVICIO_SEQUENCE.NEXTVAL,REGISTRO_VEHICULO2,7,'Pending',current_date);
        COMMIT;
        DBMS_OUTPUT.PUT_LINE('Se inserto el servicio '||INSERTA||' con NOMBRE:'||REGISTRO||' y DESCRIPCION:'||DES);
    ELSE 
        DBMS_OUTPUT.PUT_LINE('No se inserto ningun servicio, el vehiculo '||REGISTRO_VEHICULO2||' No existe');
    END IF;
 END;


EXEC PR_INSERTAR_SERVICIO(100);

--PUNTO 7 TRIGGER QUE INVOCA PR_INSERTAR_SERVICIO
CREATE or REPLACE TRIGGER TR_ACTUALIZAR_KILOMETRAJE
AFTER UPDATE OF KILOMETRAJE ON VEHICULO
FOR EACH ROW
DECLARE
   REGISTRO_VEHICULO2 NUMBER;
BEGIN

    REGISTRO_VEHICULO2:=NEW.REGISTRO_VEHICULO;
    EXEC PR_INSERTAR_SERVICIO(REGISTRO_VEHICULO2);
END;

--PUNTO 8 PROCEDIMIENTO QUE DISMINUYE EL VALOR ACTUAL DEL VEHICULO
CREATE OR REPLACE PROCEDURE PR_DISMINUIR_VALOR_VEHICULO(
TIPO_VEHICULO CLASE_VEHICULO.CLASE%TYPE) IS
REGISTRO INT;
ID_CLASE2 NUMBER;
PORCENTAJE NUMBER;
CALCULO NUMBER;
BEGIN
    SELECT COUNT(*) INTO REGISTRO FROM CLASE_VEHICULO WHERE CLASE=TIPO_VEHICULO;
    SELECT ID_CLASE_VEHICULO INTO ID_CLASE2 FROM CLASE_VEHICULO WHERE CLASE=TIPO_VEHICULO;
    IF(REGISTRO<>0) THEN
        IF(TIPO_VEHICULO='Moped' OR TIPO_VEHICULO='Motorcycle' OR TIPO_VEHICULO='Trike motorcycle ') THEN
            PORCENTAJE:=0.02;           
        ELSE      
            IF (TIPO_VEHICULO='Car' OR TIPO_VEHICULO=' Light rigid heavy vehicle' OR TIPO_VEHICULO=' Medium rigid heavy vehicle') THEN
                 PORCENTAJE:=0.03;        
            ELSE
                PORCENTAJE:=0.05; 
            END IF;
        END IF;
        UPDATE VEHICULO SET VALOR_ACTUAL=VALOR_ACTUAL-(VALOR_ACTUAL*PORCENTAJE) WHERE ID_CLASE_VEHICULO=ID_CLASE2;          
        COMMIT;           
        DBMS_OUTPUT.PUT_LINE('La cantidad de registros actualizados es:'||sql%rowcount ||' para el tipo de vehiculo:'||TIPO_VEHICULO);
    ELSE
         DBMS_OUTPUT.PUT_LINE('No se encontro el tipo de vehiculo:'||TIPO_VEHICULO);
    END IF;
END;
